import time
# иҺ·еҸ–еҪ“еүҚеӣҪйҷ… UTC ж—¶й—ҙ
datetime.utcfromtimestamp(time.time())


def now_time_get_time_offset(tmp_time=time.time()):
        """
        йҖҡиҝҮдј е…Ҙж—¶й—ҙиҺ·еҸ–дёҺUTCж—¶еҢәеҒҸз§»йҮҸ й»ҳи®ӨеҪ“еүҚж—¶й—ҙ

        :param tmp_time: type(time) еҪ“еүҚж—¶й—ҙжҲі e.g. 1669690645.691196
        :return int
        """
        time_offset = (datetime.datetime.fromtimestamp(tmp_time) -
                       datetime.datetime.utcfromtimestamp(tmp_time)).total_seconds() / 60 / 60
        return int(time_offset)

from pytz import timezone
tz = timezone("Asia/Shanghai")

def time_zone_get_time_offset(time_zone="Asia/Shanghai"):
        """
        йҖҡиҝҮдј е…Ҙж—¶еҢәиҺ·еҸ–дёҺUTCж—¶еҢәеҒҸз§»йҮҸ й»ҳи®Өзі»з»ҹж—¶еҢә

        :param time_zone: type(str) ж—¶еҢә e.g. Asia/Shanghai
        :return int
        """
        tz = timezone(time_zone)
        res = datetime.datetime.now(tz).utcoffset().total_seconds() / 60 / 60
        return int(res)
